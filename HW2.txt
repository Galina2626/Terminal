1. Сделать папку dir_1	1. mkdir dir_1
2. Зайти в папку dir_1	2. cd dir_1
3. Создать папку inner_dir_1	3. mkdir inner_dir_1
4. Посмотреть где ты находишься	4. pwd
5. Находясь в папке dir_1 создать пустой текстовый файл tf_1.txt	5. touch tf_1.txt
6. В папке dir_1 создать файл tf_2.txt со строками:	6. cat > tf_2.txt
- the first 1	the first 1
- the second 2	the second 2
- the third 3	the third 3 ENTER, Ctrl+C
7. Зайти в папку inner_dir_1	7. cd inner_dir_1
8. Через cat сделать текстовый файл tf_3.txt с любыми строками	8. cat >tf_3.txt {text} ENTER Ctrl+C
9. Через cat добавить в текстовый файл tf_3.txt строку “the second 2”	9. cat >>tf_3.txt the second 2 ENTER Ctrl+C
10. Через cat добавить в текстовый файл tf_3.txt строку “the sec 2”	10. cat >>tf_3.txt the sec 2 ENTER Ctrl+C
11. Через cat добавить в текстовый файл tf_2.txt строку “the sec 3”	11. cat >>../tf_2.txt the sec 3 ENTER Ctrl+C
12. Через cat добавить в текстовый файл tf_3.txt строку “the SeCoNd 2”	12. cat >>tf_3.txt the SeCoNd 2 ENTER Ctrl+C
13. Через cat добавить в текстовый файл tf_2.txt строку “the seConD 2”	13. cat >>../tf_2.txt the seConD 2 ENTER Ctrl+C
14. Сделать текстовый файл tf_4.txt в котором будет 15 строк.	14. cat >tf_4.txt {15 строк} ENTER Ctrl+C
15. Сделать текстовый файл tF_5.txt в котором будет 13 строк.	15. cat >tf_5.txt {13 строк} ENTER Ctrl+C
16. Вывести список всех файлов в папке.	16. ls -a
17. Выйти из папки inner_dir_1	17. cd ..
18. Вывести содержимое файла tf_3.txt в терминал.	18. cat /d/Homework2/dir_1/inner_dir_1/tf_3.txt
19. Найти путь к файлу tf_4.txt	19. find . -name tf_4.txt
20. Очистить файл tf_4.txt от содержимого без удаления самого файла.	20. >/d/Homework2/dir_1/inner_dir_1/tf_4.txt
21. Найти путь к файлам у которых есть “tf” в названии.	21. find -name 'tf*'
22. Найти путь к файлам у которых есть “tf” в названии и буквы в любом регистре.	22. find -iname 'tf*'
23. Найти строки в файлах где есть комбинация букв “sec” в текущей папке	23. grep sec *
24. Найти строки в файлах где есть комбинация букв “sec” в любом регистре в текущей папке	24. grep -i sec *
25. Найти строки в файлах где есть только комбинация букв “sec” в текущей папке	25. grep -w sec *
26. Найти строки в файлах где есть только комбинация букв “sec” в любом регистре в текущей папке	26. grep -w -i sec *
27. Найти строки в файлах где есть комбинация букв “second” в текущей папке	27. grep second *
28. Найти строки в файлах где есть комбинация букв “second” в любом регистре в текущей папке	28. grep -i second *
29. Найти строки в файлах где есть комбинация букв “second” во всех папках ниже уровнем	29. grep -r second */*.*
30. Найти только путь и название файла в строках которых есть комбинация букв “second” в текущей папке	30. grep -l second *
31. Найти все строки во всех файлах где нет комбинации “second”	31. grep -v -r second *
32. Найти только название и путь к файлам где нет комбинации “second”	32. grep -L -r second *
33. Вывести в терминал 4 последних строк любого текстового файла	33. tail -n4 /d/Homework2/dir_1/inner_dir_1/tf_4.txt
34. Вывести в терминал 4 первые строки любого текстового файла.	34.head -n4 /d/Homework2/dir_1/inner_dir_1/tf_4.txt

37. Команда в одну строку. Скопировать в любую одну папку текстовые файлы у которых в содержимом есть слово “sec”	37. grep -lr 'sec' . | xargs cp -t ./qa
38. Команда в одну строку. Найти все строки c “sec” во всех текстовых файлах, скопировать и вставить эти строки в один новый созданный текстовый файл.	38. grep -r 'sec' * . | cat >f123.txt

40. Просто вывести в терминал строку “Good job!!”	40. echo 'Good job!!'
